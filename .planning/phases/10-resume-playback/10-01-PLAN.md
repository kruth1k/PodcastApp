Add new endpoint to get the most recently played episode with position.

Implementation:
1. Add new endpoint after existing position endpoints (after line 125):
   ```python
   @router.get("/last-played", response_model=Optional[schemas.LastPlayedResponse])
   def get_last_played(
       request: Request,
       db: Session = Depends(get_db)
   ):
       user_id = get_current_user_id(request)
       
       # Get the most recent playback position (latest updated_at)
       position = db.query(models.PlaybackPosition).filter(
           models.PlaybackPosition.user_id == user_id
       ).order_by(models.PlaybackPosition.updated_at.desc()).first()
       
       if not position:
           return None
       
       # Get episode details from the podcast's episodes
       # Query through Podcast -> Episode relationship
       episode = None
       for podcast in db.query(models.Podcast).filter(models.Podcast.user_id == user_id).all():
           for ep in podcast.episodes:
               if ep.id == position.episode_id:
                   episode = ep
                   break
           if episode:
               break
       
       if not episode:
           return None
       
       return {
           "episode_id": position.episode_id,
           "position_seconds": position.position_seconds,
           "episode": {
               "id": episode.id,
               "title": episode.title,
               "audio_url": episode.audio_url,
               "podcast_id": episode.podcast_id
           }
       }
   ```

Note: The episode relationship might need adjustment based on actual model structure.
