  <action>
Create frontend data layer:

**web/src/lib/api.ts:**
```typescript
const API_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000';

export const api = {
  getPodcasts: () => fetch(`${API_URL}/api/podcasts`).then(r => r.json()),
  getPodcast: (id: string) => fetch(`${API_URL}/api/podcasts/${id}`).then(r => r.json()),
  addPodcast: (feedUrl: string) => fetch(`${API_URL}/api/podcasts`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ feedUrl })
  }).then(r => r.json()),
  deletePodcast: (id: string) => fetch(`${API_URL}/api/podcasts/${id}`, { method: 'DELETE' }),
  getEpisodes: (podcastId: string) => fetch(`${API_URL}/api/episodes?podcast_id=${podcastId}`).then(r => r.json()),
};
```
